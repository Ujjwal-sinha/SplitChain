How Civic Auth is implemented in SplitChain:

// Civic Auth Button Component
import { UserButton, useUser, SignInButton } from "@civic/auth-web3/react"

export function CivicAuthButton() {
  const { user } = useUser()

  if (user) {
    return <UserButton />
  }

  return (
    <SignInButton className="bg-green-500 text-black" />
  )
}
